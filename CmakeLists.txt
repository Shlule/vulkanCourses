# Inclure GLM
set(GLM_DIR "C:/CXXLib/glm-master/glm")  # Remplace "path_to_glm" par le chemin vers GLM
include_directories(${GLM_DIR})

add_executable(${PROJECT_NAME} main.cpp)

# Lier les biblioth√®ques
